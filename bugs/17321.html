Most likely non-PCH errors; that can be fixed by adding this block of code after sdk.h include in file

CscopePlugin.cpp

#ifndef CB_PRECOMP

#include <wx/process.h>

#include "cbeditor.h"

#include "cbproject.h"

#include "editormanager.h"

#include "logmanager.h"

#include "manager.h"

#include "projectmanager.h"

#endif
