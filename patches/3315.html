+#ifndef __WXMSW__
+void IncrementalSearch::OnMenuEditPaste(wxCommandEvent& event)
+{
+    // Process clipboard data only if we have the focus
+    if ( !IsAttached() )
+    {
+        event.Skip();
+        return;
+    }
+
+    wxWindow* pFocused = wxWindow::FindFocus();
+    if (!pFocused)
+    {
+        event.Skip();
+        return;
+    }
+
+    if (pFocused == m_pTextCtrl)
+        m_pTextCtrl->Paste();
+    else
+        event.Skip();
+}
+#endif
