+bool EditorBase::QueryClose()
+{
+    if (GetModified())
+    {
+        wxString msg;
+        msg.Printf(_("File %s is modified...\nDo you want to save the changes?"), GetFilename().c_str());
+        switch (cbMessageBox(msg, _("Save file"), wxICON_QUESTION | wxYES_NO | wxCANCEL))
+        {
+        case wxID_YES:
+            if (!Save())
+                return false;
+            break;
+        case wxID_NO:
+            break;
+        case wxID_CANCEL:
+            return false;
+        }
+        SetModified(false);
+    }
+    return true;
+}
